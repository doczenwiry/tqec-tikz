% A spider with at most four legs must have all its leg in the same plane (X, Y or Z)

%\sc Minimal length of path connecting:
%\begin{itemize}
%	\item Two adjacent spiders of the same color lying in the same plane : 1
%	\item Two adjacent spiders of different colors lying in the same plane : 3
%\end{itemize}

% Assigning plane constraints may increase the Manhattan Distance between two spiders
% This means they are not adjacent in 3D and require a sequence of edges to connect them.

\begin{document}
	\Legend{}

	% Two neighbouring RED spiders
	\ZXGraph[labels=show]{
		\Node[type=X, identifier=S1]{( 0, 0)}
		\Node[type=X, identifier=S2]{(+1, 0)}

		\Edge{S1}{S2}
	}

	% Two RED spiders in same PLANE with distances (0, 1, 0)
	% Optimal Manhattan Distance = 1. Overhead: none.
	\ZXGraph[labels=show]{
		\Node[type=X, plane=X, identifier=S1]{( 0, 0)}
		\Node[type=X, plane=X, identifier=S2]{(+1, 0)}

		\Edge[axis=Y]{S1}{S2}
	}

	% RED and BLUE spiders in same PLANE with distances (0, 1, 0)
	% Optimal Manhattan Distance = 7. Overhead: +6.
	\ZXGraph[labels=show]{
		\Node[type=X, plane=X, identifier=S1]{( 0, 0)}
		\Node[type=Z, plane=X, identifier=S2]{([shift={(-30: 1)}] nodeS1)}

		\Node[type=X, plane=X, identifier=E1]{([shift={( 90: 1)}] nodeS1)}
		\Node[type=Z, plane=Y, identifier=E2]{([shift={( 90: 1)}] nodeE1)}
		\Node[type=X, plane=Z, identifier=E3]{([shift={( 30: 1)}] nodeE2)}
		\Node[type=Z, plane=X, identifier=E4]{([shift={(-30: 1)}] nodeE3)}
		\Node[type=X, plane=Y, identifier=E5]{([shift={(-90: 1)}] nodeE4)}
		\Node[type=X, plane=Y, identifier=E6]{([shift={(210: 1)}] nodeE5)}

		\Edge[axis=Z]{S1}{E1}
		\Edge[axis=Z]{E1}{E2}
		\Edge[axis=X]{E2}{E3}
		\Edge[axis=Y]{E3}{E4}
		\Edge[axis=Z]{E4}{E5}
		\Edge[axis=X]{E5}{E6}
		\Edge[axis=Z]{E6}{S2}
	}

	\ZXGraph[style=cube]{
		\Node[type=X, plane=X, identifier=S1]{( 0, 0)}
		\Node[type=Z, plane=X, identifier=S2]{([shift={(-30: 1)}] nodeS1)}

		\Node[type=X, plane=X, identifier=E1]{([shift={( 90: 1)}] nodeS1)}
		\Node[type=Z, plane=Y, identifier=E2]{([shift={( 90: 1)}] nodeE1)}
		\Node[type=X, plane=Z, identifier=E3]{([shift={( 30: 1)}] nodeE2)}
		\Node[type=Z, plane=X, identifier=E4]{([shift={(-30: 1)}] nodeE3)}
		\Node[type=X, plane=Y, identifier=E5]{([shift={(-90: 1)}] nodeE4)}
		\Node[type=X, plane=Y, identifier=E6]{([shift={(210: 1)}] nodeE5)}

		\Edge[axis=Z]{S1}{E1}
		\Edge[axis=Z]{E1}{E2}
		\Edge[axis=X]{E2}{E3}
		\Edge[axis=Y]{E3}{E4}
		\Edge[axis=Z]{E4}{E5}
		\Edge[axis=X]{E5}{E6}
		\Edge[axis=Z]{E6}{S2}
	}

\end{document}
