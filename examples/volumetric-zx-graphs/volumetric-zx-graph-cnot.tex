% A spider with at most four legs must have all its leg in the same plane (X, Y or Z)

%\sc Minimal length of path connecting:
%\begin{itemize}
%	\item Two adjacent spiders of the same color lying in the same plane : 1
%	\item Two adjacent spiders of different colors lying in the same plane : 3
%\end{itemize}

% Assigning plane constraints may increase the Manhattan Distance between two spiders
% This means they are not adjacent in 3D and require a sequence of edges to connect them.

\begin{document}
	% CNOT
	\ZxGraph{
		\Node[type=O, identifier=B1]{(0, 0, 0)}
		\Node[type=X, identifier=X1]{(0, 1, 0)}
		\Node[type=O, identifier=B2]{(0, 0,-1)}

		\Node[type=O, identifier=B3]{(0, 2, 0)}
		\Node[type=Z, identifier=Z1]{(0, 1,-1)}
		\Node[type=O, identifier=B4]{(0, 2,-1)}

		\Edge{X1}{B1}
		\Edge{X1}{B3}
		\Edge{X1}{Z1}
		\Edge{Z1}{B2}
		\Edge{Z1}{B4}
	}

	% CNOT. With plane constraints; the two spiders are adjacent in 3D (i.e. MD=1).
	\ZxGraph{
		\Node[type=O, identifier=B1]{(0, 0, 0)}
		\Node[type=X, plane=X, identifier=X1]{(0, 1, 0)}
		\Node[type=O, identifier=B3]{(0, 2, 0)}

		\Node[type=O, identifier=B2]{( 1, 1,-1)}
		\Node[type=Z, plane=Y, identifier=Z1]{(0, 1,-1)}
		\Node[type=O, identifier=B4]{(-1, 1,-1)}

		\Edge[axis=Y]{X1}{B1}
		\Edge[axis=Y]{X1}{B3}
		\Edge[axis=Z]{X1}{Z1}
		\Edge[axis=X]{Z1}{B2}
		\Edge[axis=X]{Z1}{B4}
	}

	% CNOT. With plane constraints; the two spiders are adjacent in 3D (i.e. MD=1).
	\ZxGraph[style = bg]{
		\Node[type=O, identifier=B1]{(0, 0, 0)}
		\Node[type=X, plane=X, identifier=X1]{(0, 1, 0)}
		\Node[type=O, identifier=B3]{(0, 2, 0)}

		\Node[type=O, identifier=B2]{( 1, 1,-1)}
		\Node[type=Z, plane=Y, identifier=Z1]{(0, 1,-1)}
		\Node[type=O, identifier=B4]{(-1, 1,-1)}

		\Edge[axis=Y]{X1}{B1}
		\Edge[axis=Y]{X1}{B3}
		\Edge[axis=Z]{X1}{Z1}
		\Edge[axis=X]{Z1}{B2}
		\Edge[axis=X]{Z1}{B4}
	}

\end{document}
